Dropdown

lkDropdown(el, opts) — wraps <select> elements

Live Demo

HTML

<select id="country"></select>

JavaScript

const country = Look.lkDropdown('#country', {
  label: 'Country',
  name: 'country',
  required: true,
  items: [
    { value: '',   text: 'Select a country…' },
    { value: 'us', text: 'United States' },
    { value: 'uk', text: 'United Kingdom' },
    { value: 'jp', text: 'Japan', disabled: true },
  ],
});

country.value;           // 'us'
country.selectedIndex;   // 1
country.selectedText;    // 'United States'
country.setError('Required');
country.clearError();
country.destroy();
