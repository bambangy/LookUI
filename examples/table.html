JavaScript

const table = Look.lkTable('#data', {
  sortable: true,
  onSort({ column, order }) {
    console.log(column, order);
  },
});

table.sortBy = 2;       // sort by column index
table.sortOrder;         // 'asc' or 'desc'
table.destroy();
