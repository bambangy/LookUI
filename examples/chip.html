JavaScript

const tag = Look.lkChip('#tag', {
  label: 'JavaScript',
  removable: true,
  onRemove() { console.log('removed'); },
  onSelect(sel) { console.log('selected:', sel); },
});

tag.label;              // 'JavaScript'
tag.selected;           // boolean
tag.destroy();
