    API Quick Reference

    Creating a DataSource

    const ds = Look.lkDataSource({
  keyField: 'id',             // unique key per record
  data: [...],                // initial data array
  pageSize: 5,                // rows per page
  page: 1,                    // starting page
  filter: null,               // initial filter (see below)
  sort: null,                 // initial sort (see below)
});

    Filtering 

    // Object filter — field rules
ds.setFilter({ status: 'active' });
ds.setFilter({ age: { gte: 18, lt: 65 } });
ds.setFilter({ name: { contains: 'ada' } });
ds.setFilter({ role: { in: ['admin', 'editor'] } });

// Function filter — full control
ds.setFilter((item) => item.salary > 50000);

// Combine multiple field rules
ds.setFilter({
  status: 'active',
  department: { in: ['Engineering', 'Design'] },
  name: { contains: searchText },
});

// Clear filter
ds.clearFilter();

    Sorting 

    // Single field sort
ds.setSort({ field: 'name', dir: 'asc' });
ds.setSort({ field: 'salary', dir: 'desc' });

// Multi-field sort
ds.setSort([
  { field: 'department', dir: 'asc' },
  { field: 'salary', dir: 'desc' },
]);

// Custom comparer
ds.setSort({ comparer: (a, b) => a.name.localeCompare(b.name) });

// Clear sort
ds.clearSort();

    Paging 

    ds.setPage(2);
ds.setPageSize(25);
ds.page;          // current page (1-based)
ds.totalPages;    // computed total pages
ds.total;         // total filtered count

    CRUD Operations 

    // Add
await ds.add({ id: 99, name: 'New User' }, { sync: false });

// Update by ID or predicate
await ds.update(5, { name: 'Updated Name' }, { sync: false });
await ds.update(item => item.email === 'x@y.com', { status: 'active' });

// Remove by ID
await ds.remove(5, { sync: false });

// Replace all data
ds.set([...newData]);

// Clear all
ds.clear();

    Subscribing to Changes 

    // Subscribe — fires on every change, immediate by default
const unsub = ds.subscribe((state, evt) => {
  console.log(evt.reason);     // 'filter', 'sort', 'page', 'add', 'remove', etc.
  console.log(state.view);     // current page of visible rows
  console.log(state.total);    // total filtered count
  console.log(state.totalPages);
});

// Granular events
ds.on('add', ({ record }) => console.log('Added:', record));
ds.on('remove', ({ record }) => console.log('Removed:', record));
ds.on('filter', ({ filter }) => console.log('Filter changed'));
ds.on('sort', ({ sort }) => console.log('Sort changed'));

// Batch multiple changes (single event)
ds.batch(() => {
  ds.setFilter({ status: 'active' });
  ds.setSort({ field: 'name', dir: 'asc' });
  ds.setPage(1);
});

// Unsubscribe
unsub();

    Remote Transport (optional) 

    const ds = Look.lkDataSource({
  keyField: 'id',
  autoLoad: true,
  transport: {
    read:   { url: '/api/users', method: 'GET' },
    create: { url: '/api/users', method: 'POST' },
    update: { url: '/api/users/{id}', method: 'PUT' },
    delete: { url: '/api/users/{id}', method: 'DELETE' },
  },
  server: {
    filter: true,    // send filter to server
    sort: true,      // send sort to server
    paging: true,    // send page/pageSize to server
  },
  request: {
    headers: { Authorization: 'Bearer token' },
  },
  hooks: {
    onError: ({ error }) => console.error(error),
  },
});

// Or use a function transport
const ds2 = Look.lkDataSource({
  transport: {
    read: async (params) => {
      const res = await fetch(`/api?page=${params.page}`);
      return res.json(); // { items: [...], total: 100 }
    },
  },
});

    Lookup Helpers 

    ds.getById(5);                        // find record by key
ds.find(item => item.name === 'Ada'); // find first match
ds.items;                             // all raw (unfiltered) items
ds.view;                              // current page slice
ds.toJSON();                          // full snapshot
