Modal Title

This modal uses Look.lkModal() with focus trapping.

HTML

<button id="trigger">Open</button>
<div id="modal-container">
  <div class="lk-modal-overlay">
    <div class="lk-modal" role="dialog" aria-modal="true">
      <div class="lk-modal__header">…</div>
      <div class="lk-modal__body">…</div>
      <div class="lk-modal__footer">…</div>
    </div>
  </div>
</div>

JavaScript

const modal = Look.lkModal('#trigger', '#modal-container');
modal.open();
modal.close();
modal.destroy();
