// Dropdown toggle (add lk-btn-dropdown--open)
const dd = document.querySelector('.lk-btn-dropdown');
dd.querySelector('.lk-btn').addEventListener('click', () => {
  dd.classList.toggle('lk-btn-dropdown--open');
});
