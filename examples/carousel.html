lkCarousel(el, opts) — slide navigation with autoplay

HTML

<div id="my-carousel">
  <div class="lk-carousel__track">
    <div class="lk-carousel__slide">Slide 1</div>
    <div class="lk-carousel__slide">Slide 2</div>
  </div>
  <button class="lk-carousel__prev">&lsaquo;</button>
  <button class="lk-carousel__next">&rsaquo;</button>
  <div class="lk-carousel__indicators">
    <button class="lk-carousel__dot"></button>
    <button class="lk-carousel__dot"></button>
  </div>
</div>

JavaScript

const carousel = Look.lkCarousel('#my-carousel', {
  autoplay: true,
  interval: 3000,
  loop: true,
});

carousel.activeIndex;    // 0
carousel.next();
carousel.prev();
carousel.goTo(2);
carousel.destroy();
