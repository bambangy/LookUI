<div id="volume">
  <div class="lk-slider__track">
    <div class="lk-slider__fill"></div>

JavaScript

const vol = Look.lkSlider('#volume', {
  min: 0, max: 100, value: 50, step: 5,
  onChange(v) { console.log(v); },
});

vol.value;              // 50
vol.value = 75;
vol.min; vol.max; vol.step;
vol.destroy();
