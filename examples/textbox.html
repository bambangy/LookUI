Textbox

lkTextbox(el, opts) — wraps <input> or <textarea>

Live Demo

HTML

<input id="email" type="email" />
<textarea id="notes"></textarea>

JavaScript

// Basic — auto-creates .lk-field wrapper with label
const email = Look.lkTextbox('#email', {
  label: 'Email address',
  name: 'email',
  type: 'email',
  required: true,
});

// Get / set value
email.value;                  // current value
email.value = 'user@example.com';

// Validation
email.setError('Invalid email format');
email.clearError();

// Properties
email.label = 'Work email';  // update label text
email.required = true;
email.readonly = false;
email.enabled = false;        // disable
email.hidden = true;          // hide

// Cleanup
email.destroy();
